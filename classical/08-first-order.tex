\section{一阶系统时域分析}
\label{sec:first-order-system}

\subsection{一阶系统的传递函数}
\[G(s) = \frac{K}{Ts + 1}\]

其中 $K$ 为增益，$T$ 为时间常数。

\subsection{一阶系统的单位阶跃响应}
\[h(t) = K(1 - e^{-t/T})\]

\subsection{一阶系统的性能指标}
\begin{itemize}
    \item 调节时间：$t_s = 3T$（2\%误差带）或 $t_s = 4T$（5\%误差带）
    \item 无超调
    \item 上升时间：$t_r = 2.2T$
\end{itemize}
